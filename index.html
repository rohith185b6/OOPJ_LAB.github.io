		System.out.print("Enter no. of rows of Second matrix:\n");

		System.out.print("Enter no. of columns of second matrix:\n");
		c = s.nextInt();
		if(n!=r) //Condition for Matrix_Multiplication is not possible
		{
			System.out.println("The Multiplication of Matrix is not possible.");
		}
		else 	
		{
			int a[][] = new int[m][n]; //Matrix_1 
			int b[][] = new int[r][c]; //Matrix_2
			int d[][] = new int[m][c];
			//Rreading Matrix_1 elements
			System.out.println("Enter the elements of matrix1"); 
			for ( i= 0 ; i < m ; i++ )
			{ 
				for ( j= 0 ; j < n ;j++ )
					a[i][j] = s.nextInt();
			}
			//Reading Matrix_2 elements
			System.out.println("Enter the elements of matrix2");
			for ( i= 0 ; i < r ; i++ )
			{ 
				for ( j= 0 ; j < c ;j++ )
					b[i][j] = s.nextInt();
			}
			//Printing Matrix_1 elemtns
			System.out.println("\n\nEnterd elements of First Matrix :");
			for(i= 0;i < m;i++)
			{
				System.out.println("\n");
				for(j= 0;j < n;j++)
				{
					System.out.print(a[i][j]+"\t");
				}
			}
			//Printing Matrix_2 elements
			System.out.println("\n\nEnterd elements of Second Matrix :");
			for(i= 0;i < r;i++)
			{
				System.out.println("\n");
				for(j= 0;j < c;j++)
				{
					System.out.print(b[i][j]+"\t");
				}
			}
			//Matrix Multiplication
			for(i= 0;i < m;i++)
			{
				for(j= 0;j < c;j++)
				{	
					d[i][j] = 0;
					for(k= 0;k < r;k++)
					{
						d[i][j] = a[i][k]*b[k][j];
					}
				}
			}
			//output of MAtrix_Multiplication
			System.out.println("\n\nProduct of the Matrices are :\n");
			for(i=0;i < m;i++)
			{
				for(j= 0;j < c;j++)
				{
					System.out.print("\t"+d[i][j]);
				}
				System.out.print("\n");
			}
		}

  18. Write a java program to perform push, pop, display operations on a stack?

      Program : 
      import java.io.*;
import java.util.*;
class ST
{
	int top;
	int maxsize = 10;
	int stack[] = new int[maxsize];
	boolean push(Scanner sc)
	{
		if(top ==  maxsize - 1)
		{
			System.out.println("Overflow");
			return false;
		}
		else
		{
			System.out.println("Enter Item:");
			int item = sc.nextInt();
			stack[++top] = item;
			System.out.println("Item Pushed.");
			return true;
		}	
	}
	boolean pop()
	{
		if(top == -1)
		{
			System.out.println("Underflow");
			return false;	
		}
		else
		{
		top--;
		System.out.println("Item Poped");
		return true;
		}
	}	
	void display()
	{
		System.out.println("Printing Stack Elements:\n");
		for(int i = top; i > 0;i--)
		{
			System.out.print(stack[i]+"\t");
		}
	}
}	
	class Stackop
	{
		public static void main(String[] args)
		{
			int choice=0;
			Scanner sc = new Scanner(System.in);
			ST s = new ST();
			System.out.println("Stack operations using Arrays :");
			System.out.println("\n");
			while(choice!=4)
			{
				System.out.println("\nChoose one option:");
				System.out.println("1.Push\t2.Pop\t3.display\t4.Exit");
				System.out.println("Enter your choice:");
				choice = sc.nextInt();
				switch(choice)
				{
					case 1:s.push(sc);
							break;
					case 2:s.pop();
							break;
					case 3:s.display();
							break;
					case 4:System.out.println("Exit");
							System.exit(0);
							break;
					default:System.out.println("Invalid Option");								
				}
			}
		}
	}

  19. Write a java program to perform factorial of a number using recursion?

      Program : 
      import java.io.*;
import java.util.*;
class FCR
{
	public static void main(String arg[])
	{
		Scanner s = new Scanner(System.in);
		System.out.println("Enter the number :");
		int num = s.nextInt();
		int factorial = fact(num);
		System.out.println("Factorial of enterd Number is :"+factorial);
	}
	static int fact(int n)
	{
		int output;
		if(n == 1)
		{
			return 1;
		}
		output = fact(n-1)*n;
		return output;
	}
}
